unción que comprueba si hay posiciones libres o no en el tablero. Compara con su izquierda dcha arriba y abajo
Guardamos en variables auxiliares los posibles resultados. Y luego si son iguales o no se escribe en la posción del array de reaultados
*/
__global__ void  seguirJugando(float* tablero_d, bool* result, int nf, int nc) {
	bool aux1, aux2, aux3 = false;
	int posicion = blockIdx.x*blockDim.x + threadIdx.x;
	
		//si i es menor que 7 compara a si misma y con la de al lado
		if ((posicion %(nc-1)) != 0) {
			if (tablero_d[posicion] == 0 || tablero_d[posicion + 1] == 0 || tablero_d[posicion] == tablero_d[posicion + 1]) {
				aux1 = true;
			}
			else aux1 = false;
		// si es la columna 7, se compara consigo mismo
		}
		else {
			if (tablero_d[posicion] == 0) {
				aux1 = true;
			}
			else aux1 = false;
		}
		// si no estamos en la última fila, comparamos con el de abajo
		if (posicion % (nf - 1) !=0) {
			if (tablero_d[posicion] == 0 || tablero_d[posicion + nc] == 0 || tablero_d[posicion] == tablero_d[posicion + nf]) {
				aux2 = true;
			}
			else aux2 = false;
		}

		if (aux1 != aux2) result[posicion] = true;
		else if (aux1 == aux2 && aux2 == true) result[posicion] = true;
		else result[posicion] = false; 
}
